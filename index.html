#include "Arduino.h" #include "Arduino_BHY2.h" #include #include "Nicla_System.h" // ========================================== // --- CONFIGURATION --- // ========================================== #define DEVICE_NAME "NiclaL" // <<< METTRE "NiclaR" SUR L'AUTRE #define STEP_THRESH_VERTICAL 1.20 #define STEP_THRESH_HORIZ 0.60 #define STEP_MIN_DELAY 250 // ========================================== // --- UUIDS --- // ========================================== #define SERVICE_UUID "19B10000-0000-537E-4F6C-D104768A1214" // UUIDs de données #define CHAR_PAS_SESSION_UUID "19B10000-1001-537E-4F6C-D104768A1214" #define CHAR_PAS_TOTAL_UUID "19B10000-1002-537E-4F6C-D104768A1214" #define CHAR_VITESSE_UUID "19B10000-1003-537E-4F6C-D104768A1214" #define CHAR_BATTERY_UUID "19B10000-1004-537E-4F6C-D104768A1214" #define CHAR_ACCEL_UUID "19B10000-1005-537E-4F6C-D104768A1214" #define CHAR_ETAT_UUID "19B10000-1006-537E-4F6C-D104768A1214" // NOUVEAU : UUID de Commande (Reset, Calib, Veille) #define CHAR_CONTROL_UUID "19B10000-5001-537E-4F6C-D104768A1214" BLEService movementService(SERVICE_UUID); BLEUnsignedIntCharacteristic pasSessionChar(CHAR_PAS_SESSION_UUID, BLERead | BLENotify); BLEUnsignedIntCharacteristic pasTotauxChar(CHAR_PAS_TOTAL_UUID, BLERead | BLENotify); BLEFloatCharacteristic vitesseChar(CHAR_VITESSE_UUID, BLERead | BLENotify); BLEFloatCharacteristic batteryChar(CHAR_BATTERY_UUID, BLERead | BLENotify); BLEFloatCharacteristic accelChar(CHAR_ACCEL_UUID, BLERead | BLENotify); BLEUnsignedIntCharacteristic etatChar(CHAR_ETAT_UUID, BLERead | BLENotify); // Nouvelle Caractéristique d'écriture (Byte) BLEByteCharacteristic controlChar(CHAR_CONTROL_UUID, BLERead | BLEWrite); // ========================================== // --- VARIABLES --- // ========================================== SensorXYZ accelerometer(SENSOR_ID_LACC); SensorXYZ gravity(SENSOR_ID_GRA); float calG_x = 0, calG_y = 0, calG_z = 0; bool calibrated = false; bool enVeille = false; // État de veille float vitesseInst = 0, accel_vert_mag = 0, accel_horiz_mag = 0; unsigned long lastStepTimestamp = 0, lastCheck = 0, dernierMouvement = 0; bool isStepPeaking = false, enMouvement = false; uint32_t pasS = 0, pasT = 0; void setup() { Serial.begin(115200); nicla::begin(); nicla::leds.begin(); nicla::leds.setColor(red); BHY2.begin(NICLA_STANDALONE); accelerometer.begin(); gravity.begin(); if (!BLE.begin()) while (1); BLE.setLocalName(DEVICE_NAME); BLE.setAdvertisedService(movementService); movementService.addCharacteristic(pasSessionChar); movementService.addCharacteristic(pasTotauxChar); movementService.addCharacteristic(vitesseChar); movementService.addCharacteristic(batteryChar); movementService.addCharacteristic(accelChar); movementService.addCharacteristic(etatChar); movementService.addCharacteristic(controlChar); // Ajout BLE.addService(movementService); BLE.advertise(); delay(1000); calibrateZ(); } void loop() { BLE.poll(); BHY2.update(); // --- GESTION DES COMMANDES (TELECOMMANDE) --- if (controlChar.written()) { byte cmd = controlChar.value(); switch(cmd) { case 1: // RESET pasS = 0; pasT = 0; vitesseInst = 0; pasSessionChar.writeValue(0); pasTotauxChar.writeValue(0); etatChar.writeValue(0); break; case 2: // RECALIBRATION enVeille = false; // On se réveille pour calibrer accelerometer.begin(); gravity.begin(); calibrateZ(); break; case 3: // MISE EN VEILLE if (!enVeille) { enVeille = true; nicla::leds.setColor(off); // Eteindre LED // On garde BLE actif, mais on stop les capteurs gourmands // Note: accelerometer.end() ne marche pas toujours bien sur BHY2, // on va surtout arrêter de traiter les données. etatChar.writeValue(0); // Etat repos accelChar.writeValue(0); vitesseChar.writeValue(0); } break; case 4: // REVEIL if (enVeille) { enVeille = false; // On indique le réveil if(calibrated) nicla::leds.setColor(green); else nicla::leds.setColor(red); } break; } } // Si on est en veille, on ne fait rien de physique if (enVeille) return; if (millis() - lastCheck >= 20) { if (calibrated) { updatePhysics(); detectStepScientific(); } updateBattery(); lastCheck = millis(); } } void calibrateZ() { float sumX = 0, sumY = 0, sumZ = 0; int samples = 50; nicla::leds.setColor(blue); for(int i=0; i 0) { calG_x = avgX / mag; calG_y = avgY / mag; calG_z = avgZ / mag; calibrated = true; nicla::leds.setColor(green); } else { nicla::leds.setColor(red); } } void updatePhysics() { float scale = 9.81 / 4096.0; float raw_ax = accelerometer.x() * scale; float raw_ay = accelerometer.y() * scale; float raw_az = accelerometer.z() * scale; float vert_component = (raw_ax * calG_x) + (raw_ay * calG_y) + (raw_az * calG_z); accel_vert_mag = abs(vert_component); float horiz_ax = raw_ax - (vert_component * calG_x); float horiz_ay = raw_ay - (vert_component * calG_y); float horiz_az = raw_az - (vert_component * calG_z); accel_horiz_mag = sqrt(horiz_ax*horiz_ax + horiz_ay*horiz_ay + horiz_az*horiz_az); accelChar.writeValue(enMouvement ? accel_horiz_mag : 0.00); if (enMouvement) { float dt = 0.02; vitesseInst += (accel_horiz_mag * dt); vitesseInst *= 0.95; } else { vitesseInst *= 0.8; if(vitesseInst < 0.1) vitesseInst = 0; } vitesseChar.writeValue(vitesseInst); } void detectStepScientific() { unsigned long currentMillis = millis(); bool impactCondition = (accel_vert_mag > STEP_THRESH_VERTICAL); bool motionCondition = (accel_horiz_mag > STEP_THRESH_HORIZ); if (impactCondition && motionCondition) isStepPeaking = true; if (isStepPeaking && (accel_vert_mag < STEP_THRESH_VERTICAL)) { isStepPeaking = false; if (currentMillis - lastStepTimestamp > STEP_MIN_DELAY) registerStep(currentMillis); } if (enMouvement && (currentMillis - dernierMouvement > 2000)) { enMouvement = false; etatChar.writeValue(0); } } void registerStep(unsigned long timeMs) { unsigned long stepDuration = timeMs - lastStepTimestamp; lastStepTimestamp = timeMs; pasS++; pasT++; pasSessionChar.writeValue(pasS); pasTotauxChar.writeValue(pasT); enMouvement = true; dernierMouvement = timeMs; int type = 1; if (stepDuration < 400) type = 2; if (stepDuration < 300) type = 3; etatChar.writeValue(type); } void updateBattery() { static unsigned long tBat = 0; if (millis() - tBat > 10000) { tBat = millis(); float voltage = nicla::getCurrentBatteryVoltage(); batteryChar.writeValue(voltage); // Alerte batterie if (voltage < 3.50 && !enVeille) { nicla::leds.setColor(red); delay(50); nicla::leds.setColor(off); } } }
